{{- /* Robots meta tag for SEO */ -}} {{- if eq .Kind "404" -}} {{- else -}} {{- $robots := slice -}} {{- if .Params.noindex -}} {{- $robots = $robots | append "noindex" -}} {{- end -}} {{- if .Params.nofollow -}} {{- $robots = $robots | append "nofollow" -}} {{- end -}} {{- if .Params.noarchive -}} {{- $robots = $robots | append "noarchive" -}} {{- end -}} {{- if .Params.nosnippet -}} {{- $robots = $robots | append "nosnippet" -}} {{- end -}} {{- if eq (len $robots) 0 -}} {{- $robots = $robots | append "index" "follow" -}} {{- end -}} {{- end -}}
